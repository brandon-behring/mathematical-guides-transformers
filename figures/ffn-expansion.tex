%! no-theme
% ffn-expansion (ch06, def-tf6-ffn) ‚Äî the position-wise feed-forward network as a widen‚Üíactivate‚Üíproject.
% FFN(x) = œÉ(x W_1 + b_1) W_2 + b_2, with W_1 ‚àà R^{d√ód_ff}, W_2 ‚àà R^{d_ff√ód}, d_ff = 4d (typical), applied
% per token. Device: WIDTH-PROPORTIONAL boxes ‚Äî the hidden layer is drawn 4√ó the height of input/output, so
% the expansion is legible at a glance (height ‚àù number of units). /authoring-figures standard: relative
% positioning + tokens, white-knockout W_1/W_2 labels on horizontal arrows, edge-dim brackets on the FREE
% outer edges of x and y. The hidden box is mid-pipeline (flow on both vertical edges), so a dimension
% BRACKET there would cross a flow arrow ‚Äî its d_ff is carried by the proportional height + a text label.
\documentclass[tikz,border=8pt]{standalone}
\usepackage{transformer-figures}
\begin{document}
\begin{tikzpicture}

  % ‚îÄ‚îÄ pipeline: x (d) ‚îÄ‚îÄW_1‚îÄ‚îÄ‚ñ∂ œÉ-hidden (d_ff = 4d, drawn 4√ó taller) ‚îÄ‚îÄW_2‚îÄ‚îÄ‚ñ∂ y (d) ‚Äî all on one axis ‚îÄ‚îÄ
  \node[tfinput,  minimum height=10mm, minimum width=11mm, font=\large] (x) {$\mathbf x$};
  % same width as x/y so ONLY height varies ‚Äî height, not bulk, carries the d_ff expansion
  \node[tfstruct, minimum height=40mm, minimum width=11mm, font=\Large, right=\tfstagegap of x] (h) {$\sigma$};
  \node[tfoutput, minimum height=10mm, minimum width=11mm, font=\large, right=\tfstagegap of h] (y) {$\mathbf y$};

  \draw[tfflow] (x.east) -- (h.west) node[midway, tfoplabel] {$\mathbf W_1$ widens};
  \draw[tfflow] (h.east) -- (y.west) node[midway, tfoplabel] {$\mathbf W_2$ projects};

  % ‚îÄ‚îÄ dimensions: brackets on the free outer edges (x.left, y.right); the hidden height carries d_ff ‚îÄ‚îÄ
  \tfdimleft{x}{$d$}
  \tfdimright{y}{$d$}
  \node[tfdim, above=4.5mm of h] {$d_{\text{ff}} = 4d$};

  % ‚îÄ‚îÄ title + caption in line-free lanes ‚îÄ‚îÄ
  \node[tfrole, above=10mm of h] {Position-wise feed-forward network};
  \node[tfcap, align=center, below=7mm of h]
       {$\operatorname{FFN}(\mathbf x)=\sigma(\mathbf x\mathbf W_1+\mathbf b_1)\,\mathbf W_2+\mathbf b_2$\quad$\cdot$\quad applied to each token independently\\[3pt]
        widen $d\!\to\! d_{\text{ff}}$,\ bend by $\sigma$ (ReLU / GELU),\ project $d_{\text{ff}}\!\to\! d$\quad(typically $d_{\text{ff}}=4d$)};

\end{tikzpicture}
\end{document}
